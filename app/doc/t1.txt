### 3.4 Database Schema
```sql
-- ========================================
-- TASKS TABLE
-- ========================================
CREATE TABLE tasks (
    id TEXT PRIMARY KEY,  -- UUID
    batch_id TEXT NOT NULL,
    
    -- Video Info
    video_path TEXT NOT NULL,
    video_type TEXT CHECK(video_type IN ('video', 'short')),
    file_size INTEGER,
    
    -- Metadata
    title TEXT NOT NULL,
    description TEXT,
    tags TEXT,  -- JSON array
    category INTEGER,
    privacy TEXT CHECK(privacy IN ('public', 'unlisted', 'private')),
    thumbnail_path TEXT,
    
    -- Status
    status TEXT CHECK(status IN ('pending', 'active', 'completed', 'failed', 'paused', 'cancelled')),
    progress REAL DEFAULT 0.0,  -- 0-100
    
    -- Worker Assignment
    worker_id TEXT,
    
    -- Retry Logic
    retry_attempt INTEGER DEFAULT 0,
    max_retries INTEGER DEFAULT 5,
    last_error TEXT,  -- JSON object
    
    -- Scheduling
    scheduled_at TEXT,  -- ISO8601
    delay_seconds INTEGER,
    
    -- YouTube Result
    youtube_video_id TEXT,
    youtube_url TEXT,
    
    -- Timestamps
    created_at TEXT NOT NULL,
    started_at TEXT,
    completed_at TEXT,
    
    -- Indexes
    FOREIGN KEY (batch_id) REFERENCES batches(id) ON DELETE CASCADE
);

CREATE INDEX idx_tasks_status ON tasks(status);
CREATE INDEX idx_tasks_batch ON tasks(batch_id);
CREATE INDEX idx_tasks_worker ON tasks(worker_id);

-- ========================================
-- BATCHES TABLE
-- ========================================
CREATE TABLE batches (
    id TEXT PRIMARY KEY,  -- UUID
    name TEXT,
    
    -- Statistics
    total_tasks INTEGER DEFAULT 0,
    completed_tasks INTEGER DEFAULT 0,
    failed_tasks INTEGER DEFAULT 0,
    
    -- Status
    status TEXT CHECK(status IN ('pending', 'active', 'completed', 'paused', 'cancelled')),
    
    -- Configuration
    config TEXT,  -- JSON: worker_count, delay_range, etc.
    
    -- Timestamps
    created_at TEXT NOT NULL,
    started_at TEXT,
    completed_at TEXT
);

-- ========================================
-- WORKERS TABLE
-- ========================================
CREATE TABLE workers (
    id TEXT PRIMARY KEY,  -- worker_1, worker_2, etc.
    status TEXT CHECK(status IN ('idle', 'busy', 'stopped')),
    current_task_id TEXT,
    tasks_completed INTEGER DEFAULT 0,
    last_active_at TEXT,
    created_at TEXT NOT NULL,
    
    FOREIGN KEY (current_task_id) REFERENCES tasks(id) ON DELETE SET NULL
);

-- ========================================
-- UPLOAD_HISTORY TABLE
-- ========================================
CREATE TABLE upload_history (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    task_id TEXT NOT NULL,
    youtube_video_id TEXT NOT NULL,
    youtube_url TEXT NOT NULL,
    
    -- Metrics
    upload_duration_seconds REAL,
    file_size_bytes INTEGER,
    avg_speed_bps REAL,
    
    -- Metadata
    video_title TEXT,
    video_type TEXT,
    
    uploaded_at TEXT NOT NULL,
    
    FOREIGN KEY (task_id) REFERENCES tasks(id) ON DELETE CASCADE
);

CREATE INDEX idx_history_date ON upload_history(uploaded_at);
CREATE INDEX idx_history_youtube ON upload_history(youtube_video_id);

-- ========================================
-- SYSTEM_STATE TABLE
-- ========================================
CREATE TABLE system_state (
    key TEXT PRIMARY KEY,
    value TEXT NOT NULL,  -- JSON
    updated_at TEXT NOT NULL
);

-- Common keys:
-- - 'system.is_paused': boolean
-- - 'system.worker_count': integer
-- - 'system.active_batch_id': uuid
-- - 'api.quota_used': integer
-- - 'api.quota_reset_at': ISO8601

-- ========================================
-- LOGS TABLE
-- ========================================
CREATE TABLE logs (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    timestamp TEXT NOT NULL,
    level TEXT CHECK(level IN ('DEBUG', 'INFO', 'WARNING', 'ERROR', 'CRITICAL')),
    component TEXT,  -- ui, orchestrator, worker_1, etc.
    message_id TEXT,
    event_type TEXT,
    message TEXT NOT NULL,
    data TEXT,  -- JSON
    
    FOREIGN KEY (message_id) REFERENCES messages(id) ON DELETE SET NULL
);

CREATE INDEX idx_logs_timestamp ON logs(timestamp);
CREATE INDEX idx_logs_level ON logs(level);
CREATE INDEX idx_logs_component ON logs(component);

-- ========================================
-- MESSAGES TABLE (Optional: Message Archive)
-- ========================================
CREATE TABLE messages (
    id TEXT PRIMARY KEY,  -- msg_uuid
    timestamp TEXT NOT NULL,
    type TEXT NOT NULL,
    source TEXT NOT NULL,
    destination TEXT NOT NULL,
    correlation_id TEXT,
    payload TEXT NOT NULL,  -- JSON
    
    -- Metadata
    processed BOOLEAN DEFAULT 0,
    processed_at TEXT
);

CREATE INDEX idx_messages_type ON messages(type);
CREATE INDEX idx_messages_correlation ON messages(correlation_id);
```

def calculate_delay(base_range: tuple[int, int],
distribution: str = 'uniform') -> int:
"""
Calculate random delay with specified distribution
Args:
    base_range: (min_seconds, max_seconds)
    distribution: 'uniform' or 'normal'

Returns:
    delay in seconds
"""
min_delay, max_delay = base_range

if distribution == 'uniform':
    return random.uniform(min_delay, max_delay)
elif distribution == 'normal':
    mean = (min_delay + max_delay) / 2
    std = (max_delay - min_delay) / 6  # 99.7% within range
    delay = random.gauss(mean, std)
    return max(min_delay, min(max_delay, delay))
def exponential_backoff(attempt: int,
base_delay: int = 2,
jitter: bool = True) -> int:
"""
Calculate exponential backoff delay with optional jitter
Args:
    attempt: retry attempt number (0-indexed)
    base_delay: base delay in seconds
    jitter: add random jitter

Returns:
    delay in seconds
"""
delay = base_delay * (2 ** attempt)

if jitter:
    jitter_amount = delay * 0.2  # Â±20%
    delay += random.uniform(-jitter_amount, jitter_amount)

return max(1, int(delay))  # minimum 1 second
